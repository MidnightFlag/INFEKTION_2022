Figure 7: A TimerEvent class

package timerbean;

import java.util.*; // for EventObject

public class TimerEvent extends EventObject {
    
    private int theTimer;
    private int theCount;
    public 
    TimerEvent(Object source, 
       int timerNumber, int timerCount) {

       super(source);   
       theTimer = timerNumber;
       theCount = timerCount; 
    }
    public int getTimerNumber() {
        return theTimer;
    }
    public int getTimerCount() { 
        return theCount;
    }
}
