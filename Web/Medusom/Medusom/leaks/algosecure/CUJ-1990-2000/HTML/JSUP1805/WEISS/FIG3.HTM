public synchronized int getInterval() {
    return theInterval;
}

public synchronized void setInterval(int interval) {
    theInterval = interval;
}

public synchronized boolean isSuspended() {
    return suspended;
}

public synchronized void setSuspended(boolean flag) {
    suspended = flag;
    if (!suspended) {
        notifyAll();
    }
}
